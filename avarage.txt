grades = [[5, 3, 3, 5, 4], [2, 2, 2, 3],
          [4, 5, 5, 2], [4, 4, 3], [5, 5, 5, 4, 5]]
students = {'Johny', 'Bilbo', 'Steve', 'Khendrik', 'Aaron' }
students = ['Aaron', 'Bilbo', 'Khendrik', 'Johny', 'Steve']
# объявим функцию avg
def avg(grades):                    #def-функция, все послежующее- её определение
    return sum(grades)/len(grades)  #return- возвращает значение из функции, sum- суммирует список
dict_ = dict(zip(students, map(avg, grades)))
print(dict_)
